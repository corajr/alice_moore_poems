\PoemTitle[Edmund: Language becomes any animal I wish,]{}\label{ch:lear_ai}
\settowidth{\versewidth}{Made muscular i'th'heat of my animal}
\begin{verse}[\versewidth]
\Character{Edmund} Language becomes any animal I wish,\\*
Made muscular i'th'heat of my animal\\*
Necessities:

\qquad \qquad I stroke\\*
And it purrs, serves the purpose,\\*
Whatever purpose

Serves me.\\*
I want:

It is enough.
\end{verse}
