\PoemTitle{Sharks}
\label{ch:sharks}
\settowidth{\versewidth}{                . . . .}
\begin{verse}[\versewidth]
     Motion-\\
Less in motion\\
              Sharks

     Hover amid\\
Helixes of light\\
              They

     Are ``living\\
Fossils,'' they\\
               Sink

     To the ``green''\\
Depths\\
                . . . .
\end{verse}
