\PoemTitle{Sharks}
\label{ch:sharks}
\settowidth{\versewidth}{\hspace*{3\vgap} \ldots}
\begin{verse}[\versewidth]
\hspace*{1\vgap} Motion-\\
Less in motion\\
\hspace*{3\vgap} Sharks

\hspace*{1\vgap} Hover amid\\
Helixes of light\\
\hspace*{3\vgap} They

\hspace*{1\vgap} Are ``living\\
Fossils,'' they\\
\hspace*{3\vgap} Sink

\hspace*{1\vgap} To the ``green''\\
Depths\\
\hspace*{3\vgap} \ldots
\end{verse}
